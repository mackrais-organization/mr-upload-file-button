Description

Plugin for ajax uploading files. Development using the jQuery. Very easy to use and easy to modify for any requirements. The advantages of this plugin are that it is very fast, flexible and also free. There is also a full documentation and many examples of how to use this plugin.

Features

  • Drag and Drop area.
  • Easily customize to fit your needs.
  • You can use both a separate widget or part of the form.
  • Previews to images, (extensions to file)
  • Own gallery to view images
  • Support multi language
  • Sortable thumbnail
  • Remove thumbnail
  • Ajax upload
  • Mobile device gallery support. Needed jQueryMobile

Custom notification


$('#demo6').mrUploadFileButton({
  'preview': true,
  'disableNotification': true, // disabled default notification
  'allowedTypes': 'image/*' // only image
})
 .on('mr:fileUpload:getError',function(event, code, message, typeError){
    if(typeError !== 'ajaxError'){
      MsgError(message,5000,0)
    }
  })
  .on('mr:fileUpload:getWarning',function(event, code, message){
    MsgWarning(message,5000,0)
  })
  .on('mr:fileUpload:ajaxDone', function (data, textStatus, errorThrown) {
    if(data && data.status === 'ok'){
      MsgSuccess(data && data.message ? data.message : 'Successful uploaded',5000,0)
    else{
      MsgError(data && data.message ? data.message : 'An error occurred!',5000,0)
    }
  })
  .on('mr:fileUpload:ajaxError', function (event, jqXHR, textStatus, errorThrown) {
    MsgError('Error ' + jqXHR.status + ' : : ' + textStatus,5000,0)
   })

$('#example_edit_album_form').on('submit',function (e) {
  e.preventDefault();
  var data = normalizeSerializeArray($(this).serializeArray());
  var mrFileUploaderData = $('#demo17').data('getRequestData')();
  $.each(data, function (name,value) {
     mrFileUploaderData.append(name,value)
  });
  $.ajax({
    'url': 'http://example.com/edit_album.php',
    'type': 'POST',
    'data': mrFileUploaderData,
    'dataType': 'JSON',
    'contentType': false,
    'processData': false,
    'success': function (result) {
      // some code after save
    }
  });

});

/**
 * Normalize serialize array to format [ {name: value}, ... ]
 * @param serializeArray
 * @return {{}}
 */
function normalizeSerializeArray (serializeArray) {
  var result = {};
  $.each(serializeArray, function (index, item) {
    if (typeof result[item.name] !== 'undefined'
          && typeof result[item.name] !== 'object') {
          var tmp = result[item.name];
          result[item.name] = [];
          result[item.name].push(tmp);
          result[item.name].push(item.value);
        } else if (typeof result[item.name] === 'object') {
          result[item.name].push(item.value);
        } else if (typeof result[item.name] === 'undefined') {
          result[item.name] = item.value;
        }
      });
      return result;
    }
